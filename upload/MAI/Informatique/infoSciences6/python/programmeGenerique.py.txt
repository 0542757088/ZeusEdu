from math import *
import matplotlib.pyplot as plt

limInf = float(input ('Quelle est le borne inférieure?'))
limSup = float(input ('Quelle est la borne supérieure?'))
param3 = float(input('Déphasage entre les ondes?'))


def f(x, theta):
	y = sin(x + theta)
	return y

# préparation de la boucle
increment = 0.1
dataX = []
dataY1 = []

x = limInf
while x <= limSup:
	y1 = f(x, param3)
	dataX.append(x)
	dataY1.append(y1)
	x = x + increment

plt.plot(dataX, dataY1)
plt.show()
